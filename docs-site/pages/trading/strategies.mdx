## Building a Mean Reversion Strategy

This strategy buys oversold and sells overbought tokens.

### Step 1: Define Parameters

```python
from mtaadao.trading import BaseStrategy
import numpy as np

class MeanReversionStrategy(BaseStrategy):
    def __init__(self):
        super().__init__("MeanReversion")
        
        # Parameters
        self.rsi_period = 14
        self.rsi_oversold = 30
        self.rsi_overbought = 70
        self.position_size = 0.02  # 2% risk per trade
        
        # Tracking
        self.positions = {}
        self.trades = []
```

### Step 2: Calculate Indicators

```python
def calculate_rsi(prices, period=14):
    """Calculate Relative Strength Index"""
    deltas = np.diff(prices)
    
    seed = deltas[:period+1]
    up = seed[seed >= 0].sum() / period
    down = -seed[seed < 0].sum() / period
    
    rs = up / down if down != 0 else np.inf
    rsi = 100 - 100 / (1 + rs)
    
    for d in deltas[period+1:]:
        if d > 0:
            up = (up * 13 + d) / 14
            down = (down * 13) / 14
        else:
            up = (up * 13) / 14
            down = (-d * 13 + down) / 14
        
        rs = up / down if down != 0 else np.inf
        rsi = 100 - 100 / (1 + rs)
    
    return rsi

def analyze(self, data):
    """Generate trading signal based on RSI"""
    prices = data["close_prices"]  # Last 100 prices
    rsi = calculate_rsi(prices, self.rsi_period)
    
    if rsi < self.rsi_oversold:
        confidence = (self.rsi_oversold - rsi) / self.rsi_oversold
        return {"signal": "BUY", "confidence": confidence}
    
    elif rsi > self.rsi_overbought:
        confidence = (rsi - self.rsi_overbought) / (100 - self.rsi_overbought)
        return {"signal": "SELL", "confidence": confidence}
    
    else:
        return {"signal": "HOLD", "confidence": 0.5}
```

### Step 3: Implement Risk Management

```python
def calculate_position_size(self, account_balance, entry_price, stop_loss_price):
    """Calculate position size based on risk"""
    risk_amount = account_balance * self.position_size
    price_risk = entry_price - stop_loss_price
    
    if price_risk <= 0:
        return 0
    
    position_size = risk_amount / price_risk
    return position_size

def execute(self, signal, market_data):
    """Execute trade with risk management"""
    if signal["signal"] == "BUY":
        entry_price = market_data["price"]
        stop_loss = entry_price * 0.97  # 3% below entry
        take_profit = entry_price * 1.05  # 5% above entry
        
        size = self.calculate_position_size(
            account_balance=100_000,
            entry_price=entry_price,
            stop_loss_price=stop_loss
        )
        
        order = {
            "symbol": "MTAA/USDC",
            "side": "BUY",
            "size": size,
            "entry_price": entry_price,
            "stop_loss": stop_loss,
            "take_profit": take_profit,
            "timestamp": market_data["timestamp"]
        }
        
        self.positions[order["id"]] = order
        return order
```

### Step 4: Track Performance

```python
def record_trade(self, position, exit_price, exit_reason):
    """Record completed trade for analysis"""
    entry_price = position["entry_price"]
    profit = (exit_price - entry_price) * position["size"]
    profit_pct = (exit_price - entry_price) / entry_price * 100
    
    trade = {
        "entry_price": entry_price,
        "exit_price": exit_price,
        "size": position["size"],
        "profit": profit,
        "profit_pct": profit_pct,
        "reason": exit_reason,
        "duration": position["timestamp"] - position["entry_timestamp"]
    }
    
    self.trades.append(trade)
    return trade
```

## Testing Your Strategy

### Backtest First
```python
from mtaadao.backtesting import Backtest

backtest = Backtest(
    strategy=MeanReversionStrategy(),
    data="historical_mtaa_data.csv",
    start_date="2024-01-01",
    end_date="2025-12-31",
    initial_capital=100_000
)

results = backtest.run()
print(f"Total Return: {results['return_pct']}%")
print(f"Sharpe Ratio: {results['sharpe_ratio']}")
print(f"Max Drawdown: {results['max_drawdown']}%")
```

## Common Mistakes

❌ **Overfitting**: Optimizing too much for past data
→ Use out-of-sample testing

❌ **Survivorship Bias**: Only looking at successful tokens
→ Include all tokens, not just winners

❌ **Look-Ahead Bias**: Using future data in analysis
→ Only use data available at decision time

❌ **Ignoring Fees**: Profits eaten by transaction costs
→ Account for all fees in backtest

❌ **No Risk Management**: Not setting stops
→ Always use position sizing and stops
